Algoritmo "semnome"

Var
   operador: caracter
   primeiroNumero, segundoNumero: inteiro
   resultado: real

Inicio
   escreva("Digite o primeiro número:")
   leia(primeiroNumero)
   limpatela

   escreva("Digite o segundo número:")
   leia(segundoNumero)
   limpatela

   escreval("Escolha o operador matemático:")
   escreva("(+,-,*,/):")
   leia(operador)
   limpatela

   escolha operador
      caso "+"
      resultado <- (primeiroNumero + segundoNumero)
      caso "-"
      resultado <- (primeiroNumero - segundoNumero)
      caso "*"
      resultado <- (primeiroNumero * segundoNumero)
      caso "/"
      resultado <- (primeiroNumero / segundoNumero)
   outrocaso
      escreval("Operador matemático inválido")
   fimescolha
   
   escreval("Primeiro Número:",primeiroNumero)
   escreval("Segundo Número:",segundoNumero)
   escreval("Resultado:", resultado)





Fimalgoritmo
